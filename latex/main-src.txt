#include <Arduino.h>
#include <WiFi.h>
#include <WiFiClientSecure.h>
#include <PubSubClient.h>
#include <ArduinoJson.h>
#include <DHT.h>

#ifdef WOKWI
const char *WIFI_SSID = "Wokwi-GUEST";
const char *WIFI_PASSWORD = "";
#else
const char *WIFI_SSID = "YOUR_WIFI_SSID";
const char *WIFI_PASSWORD = "YOUR_WIFI_PASSWORD";
#endif

const char *MQTT_BROKER = "b1c8f2cc5cd4416fb74b671a407dc0ff.s1.eu.hivemq.cloud";
const int MQTT_PORT = 8883;
const char *MQTT_CLIENT_ID = "esp32-sensor-01";
const char *MQTT_USERNAME = "hivemq.webclient.1775680760112";
const char *MQTT_PASSWORD = "amy&f0VPG3c5<8,U>qAB";

const char *TOPIC_SENSOR = "iot/farm/sensor";
const char *TOPIC_CONTROL = "iot/farm/control";

#define DHT_PIN 4
#define DHT_TYPE DHT22
#define LED_PIN 5
#define SOIL_PIN 34
#define BUZZER_PIN 18

DHT dht(DHT_PIN, DHT_TYPE);
WiFiClientSecure espClient;
PubSubClient mqttClient(espClient);

unsigned long lastPublish = 0;
unsigned long lastMQTTRetry = 0;
const unsigned long PUBLISH_INTERVAL = 5000;
const unsigned long MQTT_RETRY_MS = 5000;

bool pumpActive = false;
unsigned long pumpStartTime = 0;
const unsigned long PUMP_DURATION_MS = 5000;

void toneBuzzer(int pin, int freq, int durationMs)
{
  ledcSetup(0, freq, 8);
  ledcAttachPin(pin, 0);
  ledcWrite(0, 127);
  delay(durationMs);
  ledcWrite(0, 0);
  ledcDetachPin(pin);
}

void stopPump()
{
  digitalWrite(LED_PIN, LOW);
  pumpActive = false;
}

void handleControlPayload(const char *payload, size_t len)
{
  StaticJsonDocument<256> doc;
  DeserializationError err = deserializeJson(doc, payload, len);
  if (err)
    return;

  if (doc.containsKey("pump"))
  {
    bool pumpCmd = doc["pump"].as<bool>();
    if (pumpCmd)
    {
      int durationS = doc.containsKey("duration") ? doc["duration"].as<int>() : 5;
      pumpActive = true;
      pumpStartTime = millis();
      digitalWrite(LED_PIN, HIGH);
      Serial.printf("[PUMP] Bat bom trong %d giay\n", durationS);
    }
    else
    {
      stopPump();
      Serial.println("[PUMP] Tat bom");
    }
  }
}

void mqttCallback(char *topic, byte *payload, unsigned int length)
{
  if (strcmp(topic, TOPIC_CONTROL) == 0)
  {
    char msg[length + 1];
    memcpy(msg, payload, length);
    msg[length] = '\0';
    handleControlPayload(msg, length);
  }
}

void setupWiFi()
{
  Serial.print("Connecting to WiFi: ");
  Serial.println(WIFI_SSID);

  WiFi.mode(WIFI_STA);
  WiFi.begin(WIFI_SSID, WIFI_PASSWORD);

  while (WiFi.status() != WL_CONNECTED)
  {
    delay(100);
    Serial.print(".");
  }

  Serial.println();
  Serial.print("WiFi connected! IP: ");
  Serial.println(WiFi.localIP());
}

void reconnectMQTT()
{
  Serial.print("[MQTT] Connecting...");
  if (mqttClient.connect(MQTT_CLIENT_ID, MQTT_USERNAME, MQTT_PASSWORD))
  {
    Serial.println(" Connected!");
    mqttClient.subscribe(TOPIC_CONTROL);
    Serial.printf("[MQTT] Subscribed: %s\n", TOPIC_CONTROL);
  }
  else
  {
    Serial.printf(" Failed, error: %d\n", mqttClient.state());
  }
}

void publishSensorData(float temp, float hum, int soilRaw)
{
  StaticJsonDocument<512> doc;
  doc["device"] = "ESP32-01";
  doc["temperature"] = temp;
  doc["humidity"] = hum;
  doc["soil"] = soilRaw;
  doc["timestamp"] = millis();

  char buffer[512];
  size_t n = serializeJson(doc, buffer);
  mqttClient.publish(TOPIC_SENSOR, buffer, n);

  Serial.printf("Published: Temp=%.1f C, Hum=%.1f %%, Soil=%d\n", temp, hum, soilRaw);
}

void readSensor(float &temp, float &hum, int &soilRaw)
{
  temp = dht.readTemperature();
  hum = dht.readHumidity();
  soilRaw = analogRead(SOIL_PIN);

  if (isnan(temp))
  {
    temp = random(200, 350) / 10.0;
  }
  if (isnan(hum))
  {
    hum = random(400, 800) / 10.0;
  }
}

void checkPumpTimeout()
{
  if (pumpActive && (millis() - pumpStartTime >= PUMP_DURATION_MS))
  {
    stopPump();
    Serial.println("[PUMP] Het thoi gian, tat bom");
  }
}

void setup()
{
  Serial.begin(115200);
  Serial.println("\n========================================");
  Serial.println("  ESP32 - Smart Farm IoT System");
  Serial.println("========================================");

  pinMode(LED_PIN, OUTPUT);
  digitalWrite(LED_PIN, LOW);

  pinMode(BUZZER_PIN, OUTPUT);
  digitalWrite(BUZZER_PIN, LOW);

  pinMode(SOIL_PIN, INPUT);

  dht.begin();

  espClient.setInsecure();

  setupWiFi();

  mqttClient.setServer(MQTT_BROKER, MQTT_PORT);
  mqttClient.setCallback(mqttCallback);

  Serial.println("Setup complete!");
}

void loop()
{
  if (WiFi.status() != WL_CONNECTED)
  {
    Serial.println("[WiFi] Lost connection, reconnecting...");
    setupWiFi();
  }

  if (!mqttClient.connected())
  {
    if (millis() - lastMQTTRetry >= MQTT_RETRY_MS)
    {
      lastMQTTRetry = millis();
      reconnectMQTT();
    }
  }
  mqttClient.loop();

  checkPumpTimeout();

  unsigned long now = millis();

  if (now - lastPublish >= PUBLISH_INTERVAL)
  {
    lastPublish = now;

    float temperature, humidity;
    int soilRaw;
    readSensor(temperature, humidity, soilRaw);

    Serial.println("\n--- Sensor Data ---");
    Serial.print("Temperature: ");
    Serial.print(temperature);
    Serial.println(" C");
    Serial.print("Humidity: ");
    Serial.print(humidity);
    Serial.println(" %");
    Serial.print("Soil Moisture (raw): ");
    Serial.println(soilRaw);

    if (soilRaw < 2048)
    {
      toneBuzzer(BUZZER_PIN, 2000, 300);
      Serial.println("[ALERT] Dat khong hop li - can tuoi!");
    }

    if (mqttClient.connected())
    {
      publishSensorData(temperature, humidity, soilRaw);
    }
    else
    {
      Serial.println("[MQTT] Not connected, skipping...");
    }
  }
}
